        // 本地 Mock 解析机制
        function localMockIntentParser(text) {
            const cleanText = text.toLowerCase();
            let amount = 10;
            let token = 'USDC';
            let destination = '0x3ef...9f2b (小李)';
            let network = 'Optimism';
            let operation = 'transfer';

            // 提取代币
            if (cleanText.includes('eth')) token = 'ETH';
            if (cleanText.includes('usdt')) token = 'USDT';
            if (cleanText.includes('usdc')) token = 'USDC';

            // 提取数量
            const numberMatch = cleanText.match(/(\d+(\.\d+)?)/);
            if (numberMatch) amount = parseFloat(numberMatch[0]);

            // 提取链
            if (cleanText.includes('scroll')) {
                network = 'Scroll';
                operation = 'cross_chain_transfer';
            } else if (cleanText.includes('optimism') || cleanText.includes('op')) {
                network = 'Optimism';
                operation = 'cross_chain_transfer';
            } else if (cleanText.includes('arbitrum') || cleanText.includes('arb')) {
                network = 'Arbitrum';
                operation = 'cross_chain_transfer';
            } else if (cleanText.includes('zksync')) {
                network = 'zkSync';
                operation = 'cross_chain_transfer';
            }

            return {
                amount,
                token,
                destination,
                network,
                operation
            };
        }

        // 调用真实的 Gemini API 直连服务器
        async function callGeminiAPI(userInput) {
            const systemPrompt = `You are the imToken Token Core AI Assistant. You parse transaction requests into raw JSON. 
            Output strictly the following JSON structure and NOTHING ELSE. No markdown blocks. No extra text.
            Schema:
            {
              "amount": number,
              "token": "ETH" | "USDC" | "USDT",
              "destination": "0x3ef...9f2b" or standard user friendly address abbreviation,
              "network": "Arbitrum" | "Optimism" | "Scroll" | "zkSync" | "Mainnet",
              "operation": "transfer" | "cross_chain_transfer"
            }`;

            const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${state.apiKey}`;
            const payload = {
                contents: [{ parts: [{ text: userInput }] }],
                systemInstruction: { parts: [{ text: systemPrompt }] }
            };

            let delay = 1000;
            for (let i = 0; i < 3; i++) {
                try {
                    const response = await fetch(url, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify(payload)
                    });
                    if (!response.ok) throw new Error(`HTTP Error: ${response.status}`);
                    const data = await response.json();
                    const text = data.candidates?.[0]?.content?.parts?.[0]?.text;
                    const cleanJson = text.replace(/```json/g, '').replace(/```/g, '').trim();
                    return JSON.parse(cleanJson);
                } catch (e) {
                    if (i === 2) throw e;
                    await new Promise(r => setTimeout(r, delay));
                    delay *= 2;
                }
            }
        }
